// PAGE SWITCHING
function goToPage(n) {
  document.querySelectorAll('.page').forEach(p => p.classList.remove('show'));
  document.getElementById(`page${n}`).classList.add('show');
  if (n === 3) startParty();
}

// Generate flying hearts (Page 1)
function flyingHearts() {
  const heart = document.createElement('div');
  heart.className = 'heart';
  heart.innerHTML = '‚ù§Ô∏è';
  heart.style.left = Math.random() * 100 + 'vw';
  heart.style.animationDuration = (3 + Math.random() * 3) + 's';
  document.body.appendChild(heart);

  setTimeout(() => heart.remove(), 6000);
}
setInterval(flyingHearts, 400);

// Party effect + colorful hearts (Page 3)
function startParty() {
  setInterval(() => {
    const conf = document.createElement('div');
    conf.className = 'confetti';
    conf.style.left = Math.random() * 100 + 'vw';
    conf.style.background = `hsl(${Math.random() * 360}, 80%, 60%)`;
    conf.style.width = conf.style.height = (5 + Math.random() * 10) + 'px';
    document.body.appendChild(conf);
    setTimeout(() => conf.remove(), 3000);
  }, 150);
}
